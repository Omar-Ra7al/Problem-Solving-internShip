//Exercise: 3 Reverse a String
function reverseString(str) {
const arrOfStr = str.split("");
let reversedStr = "";
for (let i = arrOfStr.length - 1; i >= 0; i--) {
reversedStr += arrOfStr[i];
}
return reversedStr;
}
